문제 5: 간단한 계산기

목표: 두 숫자를 입력받아서 계산 결과를 보여주세요.



💡 힌트:

function 계산하기(연산자) {
    let 첫번째 = Number(document.getElementById("숫자1").value);
    let 두번째 = Number(document.getElementById("숫자2").value);
    let 결과 = 0;
    
    if (연산자 === '+') {
        결과 = 첫번째 + 두번째;
    } else if (연산자 === '-') {
        결과 = 첫번째 - 두번째;
    } else if (연산자 === '*') {
        결과 = 첫번째 * 두번째;
    }
    
    document.getElementById("계산결과").innerHTML = 
        첫번째 + " " + 연산자 + " " + 두번째 + " = " + 결과;
}

문제 7: 글자 스타일 변경하기

문제 8: 현재 시간 보여주기

목표: 버튼을 클릭하면 현재 시간이 나타나게 하세요.

💡 힌트:

function 시간보여주기() {
    let 지금 = new Date();
    let 시간정보 = 지금.getFullYear() + "년 " + 
                 (지금.getMonth() + 1) + "월 " + 
                 지금.getDate() + "일 " +
                 지금.getHours() + "시 " + 
                 지금.getMinutes() + "분";
    
    document.getElementById("시간표시").innerHTML = 시간정보;
}

문제 9: 숨기기/보이기 기능

목표: 버튼을 클릭하면 아래 박스가 사라졌다 나타났다 하게 하세요.

👋 안녕하세요! 저를 숨겨보세요!

💡 힌트:

let 보임상태 = true;

function 토글기능() {
    let 박스 = document.getElementById("토글박스");
    
    if (보임상태 === true) {
        박스.style.display = "none";
        보임상태 = false;
    } else {
        박스.style.display = "block";
        보임상태 = true;
    }
}

문제 10: 색상 맞추기 미니게임

목표: 제시된 색상과 같은 버튼을 클릭하는 게임을 만드세요.

🎯 빨간색 버튼을 찾아서 클릭하세요!

💡 힌트:

let 정답색상 = "red";

function 게임체크(선택색상) {
    if (선택색상 === 정답색상) {
        document.getElementById("게임결과").innerHTML = "🎉 정답입니다!";
    } else {
        document.getElementById("게임결과").innerHTML = "❌ 틀렸습니다. 다시 시도해보세요!";
    }
}
