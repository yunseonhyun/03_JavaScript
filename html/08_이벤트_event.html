이벤트 e, event, value, target

        1. e와 event는 같은 것
        e와 event는 모두 이벤트 객체
        
        function 기능1번(e) {
            // 기능 1번에 해당하는 기능 내용들
        }
        function 기능2번(event) {
            // 기능 2번에 해당하는 기능 내용들
        }
예제 1: e와 event는 동일하다.
            function 클릭처리(e) {
                console.log("e 사용 : ", e.type);
            }

            function 클릭처리2(event) {
                console.log("event 사용 : ", event.type);
            }
버튼을 클릭해보세요.
예제 2 : target 사용법
            function 타겟확인(e) {
                console.log("클릭된 요소 : ", e.target.tagName);
                console.log("요소 내용 : ", e.target.tagContent);

            }
스팬 요소
요소를 클릭해보세요
예제 3 : value 사용법
            function 값확인(e) {
                console.log("입력값 : ", e.target.value);
                console.log("입력 타입 : ", e.target.type);
            }
입력하거나 선택해보세요.
예제 4: 종합 활용
function 로그인처리(e) {
    e.preventDefault(); // 폼 제출 방지
    const 사용자명 = document.getElementById("username").value;   
    const 비밀번호 = document.getElementById("password").value;
    
    console.log("폼 제출 이벤트 : " e.type);
    console.log("제출 버튼 : " e.target);
    console.log("사용자명 : " 사용자명);
}
