if(만약에) - else if(아니라면 이렇게, 저렇게) - else(모두 아니면)

        
            if(조건식1) {
                조건식1이 true인 경우에만 실행
            } else if(조건식2) {
                앞선 if의 조건식이 false이면서 
                else if의 조건식이 true인 경우;
            }elss {
                모든 if, else if가 false인 경우 실행
            }
        

        * else if 구문은 원하는만큼 작성 가능

        * 앞선 if / else if 문 조건이 false인 경우
           이후 else if에서 해당 범위는 제외
